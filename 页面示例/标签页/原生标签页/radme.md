标签页结构骨架约定：并不一定是真实的DIV组合，只是为了说明一些规则。
```html

<div id="{{uuid}}">
	<!-- 选项卡组 -->
	<!-- 选项卡最临近的 必须要有 role='tablist'属性标识-->
	<div role="tablist">
		<!-- 选项卡 id 规则： 必须以 `tab-`开头 -->
		<div id="tab-123123"></div>
		<div id="pane-333"></div>
		<!-- 选项卡结束 -->
	</div>
	<!-- 选项卡组 结束-->
	
	<!-- 面板组-->
	<div>
		<!-- 面板 id 规则： 必须以`pane-` 开头，而且对应的id应该是 选项卡对应的id 属性值 -->
		<div id="pane-123123"></div>
		<div id="pane-333"></div>
		<!-- 面板 结束-->
	</div>
	<!-- 面板组 结束-->
</div>

```
